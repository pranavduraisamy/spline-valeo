def length(x):
    df=pd.DataFrame(x,columns=['x','y','z'])
    disc_dst=np.sqrt((df['x'].diff()**2) + (df['y'].diff()**2) + (df['z'].diff()**2))
    lngth=disc_dst.sum()
    return lngth

def plot_curvature(siml,spl4,spl5, prt):
    def curvature(df):
        x = df['X'].values
        y = df['Y'].values
        z = df['Z'].values
        dx = np.gradient(x)
        dy = np.gradient(y)
        dz = np.gradient(z)
        ddx = np.gradient(dx)
        ddy = np.gradient(dy)
        ddz = np.gradient(dz)
        cross = np.cross(np.vstack((dx,dy,dz)).T, np.vstack((ddx,ddy,ddz)).T)
        cross_norm = np.linalg.norm(cross, axis=1)
        first_norm = np.linalg.norm(np.vstack((dx,dy,dz)).T, axis=1)
        curvature = cross_norm / (first_norm**3)
        return curvature
    fig2=go.Scatter(x=[i for i in range(1,101,1)], y=curvature(siml),mode='lines',name='Simulated',line=dict(color='red',width=3))
    fig24=go.Scatter(x=[i for i in range(1,101,1)], y=curvature(spl4),mode='lines',name='CW10',line=dict(color='salmon',width=3))
    fig25=go.Scatter(x=[i for i in range(1,101,1)], y=curvature(spl5),mode='lines',name='CCW10',line=dict(color='darkred',width=3))
    fig3=go.Scatter(x=[i for i in range(1,101,1)], y=curvature(prt),mode='lines',name='Prototype',line=dict(color='dodgerblue',width=3))
    return fig2,fig3,fig24,fig25
