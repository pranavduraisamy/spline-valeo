Interval Halving (From Assignment 2)

function [a,b,l,k] = inthalving(f,a,b,e)
    if nargin < 5
        e = 0.1;
    end

    xm = (a+b)/2;
    l = b-a;
    k = 0;

    while abs(l) > e
        x1 = a + (l/4);
        x2 = b - (l/4);

        if f(x1) < f(xm)
            b = xm;
            xm = x1;
        elseif f(x2) < f(xm)
            a = xm;
            xm = x2;
        else
            a = x1;
            b = x2;
        end

        l = b-a;
        k = k+1;
    end
end

Exponential Search (From Assignment 2)

function [a,b,k] = expsearch(f, x_init, d, max_iter)
    if nargin < 4
        max_iter = 100;
    end
    k = 0;
    if f(x_init - abs(d)) >= f(x_init) && f(x_init) >= f(x_init + abs(d))
        d = abs(d);
    elseif f(x_init - abs(d)) <= f(x_init) && f(x_init) <= f(x_init + abs(d))
        d = -abs(d);
    else
        disp("choose diff x_init or d")
        return
    end
    x_k = x_init;
    for i = 1:max_iter
        x_k1 = x_k + (2^k) * d;

        if f(x_k1) < f(x_k)
            k = k + 1;
            x_k = x_k1;
        else
            x_prev = x_k - (2^(k-1)) * d;
            a = min(x_prev, x_k1);
            b = max(x_prev, x_k1);
            return
        end
    end
    disp("max iters")
    a = min(x_prev, x_k1);
    b = max(x_prev, x_k1);
end

Wrapper - Combining Exponential Search and Interval Halving

function [a,b,l,k_es,k_ih] = wrapper(f, x_init, d, t, max_iter, e)

    if nargin < 5
        max_iter = 100;
    end

        e = 0.1;
    end

    [a_es, b_es, k_es] = expsearch(f, x_init, d, max_iter);

end

function f = myfunction(x)

end

Derivative at any point

function g = derivative(f, x)
    x = x(:);
    n = length(x);
    g = zeros(n,1);
    h = 1e-6;
    for i = 1:n
        x_fwd = x;
        x_bwd = x;
        x_fwd(i) = x_fwd(i) + h;
        x_bwd(i) = x_bwd(i) - h;
        g(i) = (f(x_fwd) - f(x_bwd)) / (2*h);
    end
end

function [x_opt, f_opt, iter] = descent(x0, M, e1, e2)
    if nargin < 2, M = 100; end
    if nargin < 3, e1 = 1e-4; end
    if nargin < 4, e2 = 1e-4; end
    x_k = x0(:);
    k = 0;

    fprintf('\n%-5s | %-26s | %-10s | %-26s | %-10s | %-5s\n', 'iter', 'x_k', 'f(x_k)', 's', 'alpha*', 'n_alpha*');
    fprintf('%s\n', repmat('-', 1, 100)); 
    
    while true
        grad = derivative(@myfunction, x_k);
        f_val = myfunction(x_k); 
        if norm(grad) <= e1
            fprintf('%s\n', repmat('-', 1, 100));
            fprintf('Terminated: ||grad|| (%.6f) <= eps1 at iter %d\n', norm(grad), k);
            break;
        elseif k >= M
            fprintf('%s\n', repmat('-', 1, 100));
            fprintf('Terminated; Max iters\n');
            break;
        end
        
        S = -grad; 
        h = @(alpha) myfunction(x_k + alpha * S);
        alpha_init = 0;
        delta_alpha = 0.1; 
        dummy_t = 0;
        [a, b, l, k_es, k_ih] = wrapper(h, alpha_init, delta_alpha, dummy_t, 100, e2);
        alpha_opt = (a + b) / 2;
        x_str = sprintf('[%6.4f, %6.4f, %6.4f]', x_k(1), x_k(2), x_k(3));
        S_str = sprintf('[%6.4f, %6.4f, %6.4f]', S(1), S(2), S(3));
        
        fprintf('%-5d | %-26s | %-10.4f | %-26s | %-10.4f | %-5d\n', k, x_str, f_val, S_str, alpha_opt, k_es+k_ih+2);
        
        x_next = x_k + alpha_opt * S;
        
        if norm(x_next - x_k) / (norm(x_k) + eps) <= e1
            fprintf('%s\n', repmat('-', 1, 90));
            fprintf('Terminated; x <= eps1 at iter %d\n', k);
            x_k = x_next;
            break;
        end
        
        x_k = x_next;
        k = k + 1;
    end
    x_opt = x_k;
    f_opt = myfunction(x_opt);
    iter = k;
    
    fprintf('\n--- Final Optimization Results ---\n');
    fprintf('Optimal Point: [%.6f; %.6f; %.6f]\n', x_opt(1), x_opt(2), x_opt(3));
    fprintf('Minimum Value: %.6f\n', f_opt);
end
